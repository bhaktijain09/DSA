pseudocode - 

start
   
SET C = 1        #about to take 1st score ie 1st exam score 
SET Sum = 0      #no scores added yet

while C<=25 do    #while condition do         Unless and until the no. of exam scores count are less than or equal to 25, 
input S           #statement                  keep inserting score ie actual marks
Sum= Sum+S        #statement                  keep adding current score in previous sum
C=C+1             #statement                  keep incrementing no. of exam score by 1 in previous count till we reach 25

end while
Avg=Sum/25   #final logic
print Avg    #final output statement

exit
